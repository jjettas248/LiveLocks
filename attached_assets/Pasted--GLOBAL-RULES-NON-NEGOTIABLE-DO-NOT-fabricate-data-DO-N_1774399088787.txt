export function isFiniteNumber(value: unknown): value is number {
  return typeof value === "number" && Number.isFinite(value);
}

export function hasAtLeastOneValidOddsSide(
  odds?: { overOdds?: number | null; underOdds?: number | null } | null
): boolean {
  if (!odds) return false;
  return isFiniteNumber(odds.overOdds) || isFiniteNumber(odds.underOdds);
}

export function hasRealOdds(market: any): boolean {
  if (!market) return false;

  if (!isFiniteNumber(market.line)) return false;

  if (!market.odds) return false;

  if (!hasAtLeastOneValidOddsSide(market.odds)) return false;

  return true;
}

export function isFresh(timestamp?: string | number | Date | null): boolean {
  if (!timestamp) return false;

  const ms = new Date(timestamp).getTime();
  if (!Number.isFinite(ms)) return false;

3. ENGINE DEDUP + LOCK (SERVER)

const LAST_RUN = new Map<string, number>();
const DEDUP_MS = 30_000;

function shouldSkip(gameId: string): boolean {
  const now = Date.now();
  const last = LAST_RUN.get(gameId);

  if (last && now - last < DEDUP_MS) {
    return true;
  }

  LAST_RUN.set(gameId, now);
  return false;
}

export async function runLiveGameOrchestrator(gameId: string) {

    signalLocked: hasEdge ? true : Boolean(game.signalLocked),
  };
}

function resolveRenderState(game: any): RenderState {
  if (!game.awayTeam || !game.homeTeam) return "INVALID";

  const market = game.market;

  if (!hasRealOdds(market)) {
    return game.signalLocked ? "NO_SIGNAL" : "PREVIEW";
  }

  if (!canShowSignal(market)) {
    return "NO_SIGNAL";
  }

  if (hasEdge(market)) {
    return "SIGNAL";
  }

  return "NO_SIGNAL";
}

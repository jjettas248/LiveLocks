// ==============================
// TASK #93 — DERIVED MARKET HARDENING (GOLD)
// ==============================

// Helper: detect if any REAL sportsbook-backed total exists
const hasRealFullTotal = markets.full_total?.some(
  (m) => m.available && m.isDerived !== true
);

// Apply hierarchy + signal floor rules
if (markets.full_total) {
  for (const market of markets.full_total) {

    // Only operate on derived markets
    if (market.isDerived === true) {

      // ----------------------------------------
      // RULE 1 — HARD DEPRIORITIZATION
      // If ANY real market exists, derived cannot compete
      // ----------------------------------------
      if (hasRealFullTotal) {
        market.available = false;

        console.log(
          `[NCAAB DERIVED BLOCKED] game=${gameId} reason=real_market_exists`
        );

        continue;
      }

      // ----------------------------------------
      // RULE 2 — MINIMUM CONFIDENCE FLOOR
      // Kill weak derived edges
      // ----------------------------------------
      const edgeFrom50 = Math.abs((market.modelProb ?? 50) - 50);

      if (edgeFrom50 < 12) {
        market.available = false;

        console.log(
          `[NCAAB DERIVED FILTERED] game=${gameId} reason=low_confidence edge=${edgeFrom50.toFixed(
            2
          )}`
        );

        continue;
      }

      // ----------------------------------------
      // RULE 3 — FINAL SAFETY (DIRECTIONAL SANITY RECHECK)
      // (Defensive redundancy — protects against upstream drift)
      // ----------------------------------------
      if (
        (market.side === "OVER" && market.projection <= market.bookLine) ||
        (market.side === "UNDER" && market.projection >= market.bookLine)
      ) {
        market.available = false;
        market.side = null;

        console.log(
          `[NCAAB DERIVED INVALIDATED] game=${gameId} reason=projection_mismatch`
        );

        continue;
      }

      // ----------------------------------------
      // RULE 4 — FINAL CLAMP (OPTIONAL BUT SAFE)
      // Prevent any inflated derived confidence from slipping through
      // ----------------------------------------
      if (market.modelProb != null) {
        market.modelProb = Math.max(
          35,
          Math.min(65, market.modelProb)
        );
      }
    }
  }
}

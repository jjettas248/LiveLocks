const STALE_MS = 120000;

function isFresh(ts?: string | null): boolean {
  if (!ts) return false;

  const ms = new Date(ts).getTime();
  if (!Number.isFinite(ms)) return false;

  return Date.now() - ms <= STALE_MS;
}

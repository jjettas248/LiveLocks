  const calculateNBAProjection = (
    currentStat,
    minutesPlayed,
    totalMinutes,  // 48 regular, adjust for OT
    seasonAvgPerGame,
    statType
  ) => {
    const minutesRemaining = totalMinutes - minutesPlayed
    if (minutesRemaining <= 0) return currentStat
    
    // Observed rate: current pace
    const observedRate = minutesPlayed > 0
      ? currentStat / minutesPlayed
      : 0
    
    // Season baseline rate
    const avgMinutesPerGame = 32 // league avg starter
    const baselineRate = seasonAvgPerGame / avgMinutesPerGame
    
    // Blend: 70% observed + 30% baseline per PRD
    const blendedRate = 
      (observedRate * 0.70) + (baselineRate * 0.30)
    
    const projectedAdditional = 
      blendedRate * minutesRemaining
    
    return currentStat + projectedAdditional
  }

FIX 2 — GARBAGE TIME REDUCTION IF MISSING:

  const applyGarbageTimeReduction = (
    projection,
    spread,          // current live spread
    period,
    clockMinutes,
    statType
  ) => {
    // Only applies in Q4 last 5 minutes
    if (period < 4) return projection
    if (clockMinutes > 5) return projection
    
    // Threshold: spread > 15 points = likely garbage time
    if (Math.abs(spread) < 15) return projection
    
    // Reduction: starters sit, stats dry up
    // Scale reduction with spread magnitude and time
    const spreadFactor = Math.min(
      (Math.abs(spread) - 15) / 20, 1.0
    )  // 0 at 15pt, 1.0 at 35pt+
    
    const timeFactor = (5 - clockMinutes) / 5
    // 0 at 5min, 1.0 at 0:00
    
    // Max 40% reduction at garbage time extreme
    const reductionPct = 
      spreadFactor * timeFactor * 0.40
    
    console.log(
      `[GARBAGE TIME] spread=${spread} ` +
      `reduction=${(reductionPct * 100).toFixed(1)}%`
    )
    
    return projection * (1 - reductionPct)
  }

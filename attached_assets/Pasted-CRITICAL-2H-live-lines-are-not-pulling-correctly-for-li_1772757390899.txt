  const fetch2HLines = async (gameId, game) => {
    const homeTeam = game.homeTeam
    const awayTeam = game.awayTeam
    const results = {}
    
    // ATTEMPT 1: Odds API h2 markets
    // These markets go live at halftime
    try {
      const h2Markets = [
        "h2_totals",
        "h2_spreads", 
        "alternate_totals"  // fallback
      ].join(",")
      
      const res = await fetch(
        `https://api.the-odds-api.com/v4/` +
        `sports/basketball_ncaab/events/` +
        `${gameId}/odds` +
        `?apiKey=${process.env.ODDS_API_KEY}` +
        `&regions=us&markets=${h2Markets}` +
        `&oddsFormat=american`
      )
      
      if (res.ok) {
        const data = await res.json()
        
        data.bookmakers?.forEach(book => {
          book.markets?.forEach(mkt => {
            
            if (mkt.key === "h2_totals" || 
                mkt.key === "2h_totals") {
              const over = mkt.outcomes
                ?.find(o => o.name === "Over")
              const under = mkt.outcomes
                ?.find(o => o.name === "Under")
              if (over?.point) {
                results.h2Total = over.point
                results.h2OverPrice = over.price
                results.h2UnderPrice = under?.price
                results.h2TotalSource = "odds_api"
              }
            }
            
            if (mkt.key === "h2_spreads" ||
                mkt.key === "2h_spreads") {
              const lines = mkt.outcomes
                ?.map(o => o.point)
                .filter(p => p !== undefined)
              if (lines?.length) {
                results.h2Spread = lines[0]
                results.h2SpreadSource = "odds_api"
              }
            }
          })
        })
      }
    } catch (err) {
      console.warn("[2H ODDS] Failed:", err.message)
    }
    
    // ATTEMPT 2: ActionNetwork 2H
    // period=2 returns 2H lines at halftime
    try {
      const anRes = await fetch(
        `https://api.actionnetwork.com/web/v1/` +
        `scoreboard/ncaab?period=2` +
        `&bookIds=15,30,76,123,69,68`,
        { headers: { 
          "User-Agent": "Mozilla/5.0",
          "Referer": "https://www.actionnetwork.com"
        }}
      )
      if (anRes.ok) {
        const anData = await anRes.json()
        const match = anData.games?.find(g =>
          g.teams?.some(t =>
            t.display_name?.toLowerCase()
              .includes(
                homeTeam.toLowerCase().split(" ").pop()
              )
          )
        )
        if (match) {
          const odds = match.odds?.[0]
          if (odds?.total && !results.h2Total) {
            results.h2Total = odds.total
            results.h2OverPrice = odds.ml_over
            results.h2UnderPrice = odds.ml_under
            results.h2Spread = odds.spread
            results.h2TotalSource = "action_network"
          }
          results.h2OverPct = odds?.over_bets_pct
          results.h2UnderPct = odds?.under_bets_pct
        }
      }
    } catch (err) {
      console.warn("[2H AN] Failed:", err.message)
    }
    
    // ATTEMPT 3: Derive from first-half data
    // If no live 2H line found, project from H1
    if (!results.h2Total) {
      const h1Score = game.h1HomeScore + 
                      game.h1AwayScore
      const fullLine = game.odds?.overUnder
      
      if (h1Score && fullLine) {
        // 2H projected = full line - H1 actual
        // Adjusted: regression toward H1 pace
        const h1Remaining = fullLine - h1Score
        const h1Pace = h1Score / 20  // pts per min
        const h2ProjectedPace = 
          h1Pace * 0.95  // slight regression
        const h2ProjectedTotal = 
          h2ProjectedPace * 20
        
        results.h2Total = parseFloat(
          // Round to nearest 0.5
          (Math.round(h2ProjectedTotal * 2) / 2)
          .toFixed(1)
        )
        results.h2TotalSource = "derived_h1_pace"
        
        console.log("[2H DERIVED]", {
          h1Score,
          fullLine,
          h2ProjectedTotal: results.h2Total
        })
      }
    }
    
    console.log("[2H LINES] Result:", results)
    return results
  }

STEP 3 — 2H ENGINE PROBABILITY

  const calc2HEngineProb = (
    game, h2Lines, enriched
  ) => {
    const h2Total = h2Lines?.h2Total
    if (!h2Total) return null
    
    const h1HomeScore = game.h1HomeScore ?? 0
    const h1AwayScore = game.h1AwayScore ?? 0
    const h1Total = h1HomeScore + h1AwayScore
    
    // Full game projections
    const fullProj = enriched?.engine?.projTotal
    const h2Proj = fullProj 
      ? fullProj - h1Total 
      : null
    
    if (!h2Proj) {
      // Fallback: use H1 pace for H2 projection
      // H2 typically slightly slower than H1
      const h1PerMin = h1Total / 20
      const h2PerMin = h1PerMin * 0.95
      const h2ProjFallback = h2PerMin * 20
      
      const diff = h2ProjFallback - h2Total
      const rawProb = 50 + (diff * 4.0)
      
      return {
        overProb: parseFloat(Math.min(
          Math.max(rawProb, 25), 75
        ).toFixed(1)),
        underProb: parseFloat(Math.min(
          Math.max(100 - rawProb, 25), 75
        ).toFixed(1)),
        h2Proj: parseFloat(
          h2ProjFallback.toFixed(1)
        ),
        source: "h1_pace_model"
      }
    }
    
    const diff = h2Proj - h2Total
    
    // Use composite engine certainty if available
    const sourceCount = 
      enriched?.engine?.sourceCount ?? 1
    const multiplier = Math.min(
      3.5 + (sourceCount * 0.5), 7.0
    )
    
    const rawProb = 50 + (diff * multiplier)
    const overProb = parseFloat(Math.min(
      Math.max(rawProb, 20), 80
    ).toFixed(1))
    
    // Edge vs book implied probability
    const mlToProb = (ml) => ml
      ? ml < 0
        ? parseFloat(
            (Math.abs(ml) / (Math.abs(ml) + 100) * 100)
            .toFixed(1)
          )
        : parseFloat(
            (100 / (ml + 100) * 100).toFixed(1)
          )
      : null
    
    const bookOverImplied = mlToProb(
      h2Lines.h2OverPrice
    )
    const bookUnderImplied = mlToProb(
      h2Lines.h2UnderPrice
    )
    const overEdge = bookOverImplied
      ? parseFloat(
          (overProb - bookOverImplied).toFixed(1)
        )
      : null
    const underEdge = bookUnderImplied
      ? parseFloat(
          ((100 - overProb) - bookUnderImplied)
          .toFixed(1)
        )
      : null
    
    return {
      overProb,
      underProb: parseFloat(
        (100 - overProb).toFixed(1)
      ),
      h2Proj: parseFloat(h2Proj.toFixed(1)),
      overEdge,
      underEdge,
      bookOverImplied,
      bookUnderImplied,
      hasEdge: (overEdge ?? 0) >= 5 || 
               (underEdge ?? 0) >= 5,
      edgeSide: (overEdge ?? 0) >= 5 
        ? "OVER" 
        : (underEdge ?? 0) >= 5 
          ? "UNDER" 
          : null,
      source: "composite_2h"
    }
  }

  const MarketRow = ({
    label, line, 
    overProb, underProb,
    overBookImplied, underBookImplied,
    overPrice, underPrice,
    singleSide = false,
    coverProb, bookImplied, price,
    isEstimated = false,
    direction
  }) => {
    if (!line) return (
      <div style={{
        padding: "14px 20px",
        borderBottom: "1px solid #1f1f1f",
        display: "flex",
        justifyContent: "space-between",
        alignItems: "center"
      }}>
        <span style={{
          color: "#3f3f46",
          fontSize: "11px",
          textTransform: "uppercase",
          letterSpacing: "0.05em"
        }}>
          {label}
        </span>
        <span style={{
          color: "#3f3f46",
          fontSize: "12px"
        }}>
          No line
        </span>
      </div>
    )
    
    // Calculate edge vs book
    const calcEdge = (engineProb, bookImplied) => {
      if (!engineProb || !bookImplied) return null
      return parseFloat(
        (engineProb - bookImplied).toFixed(1)
      )
    }
    
    const overEdge = calcEdge(
      overProb, overBookImplied
    )
    const underEdge = calcEdge(
      underProb, underBookImplied
    )
    const spreadEdge = calcEdge(
      coverProb, bookImplied
    )
    
    // Edge color
    const edgeColor = (edge) => {
      if (!edge) return "#71717a"
      if (edge >= 5) return "#00d4aa"
      if (edge >= 2) return "#f59e0b"
      if (edge <= -5) return "#ef4444"
      return "#71717a"
    }
    
    if (singleSide) {
      return (
        <div style={{
          padding: "14px 20px",
          borderBottom: "1px solid #1f1f1f",
          background: spreadEdge >= 5
            ? "rgba(0,212,170,0.04)" : "#111111"
        }}>
          <div style={{
            display: "flex",
            justifyContent: "space-between",
            alignItems: "center"
          }}>
            <div>
              <div style={{
                color: "#71717a",
                fontSize: "11px",
                textTransform: "uppercase",
                letterSpacing: "0.05em"
              }}>
                {label}
                {isEstimated && (
                  <span style={{
                    color: "#f59e0b",
                    fontSize: "10px",
                    marginLeft: "6px"
                  }}>
                    ~est
                  </span>
                )}
              </div>
              {bookImplied && (
                <div style={{
                  color: "#52525b",
                  fontSize: "11px",
                  marginTop: "2px"
                }}>
                  Book: {bookImplied}% implied
                  {price && ` (${price > 0 
                    ? "+" : ""}${price})`}
                </div>
              )}
            </div>
            <div style={{ textAlign: "right" }}>
              <div style={{
                color: "#ffffff",
                fontSize: "18px",
                fontWeight: "700"
              }}>
                {line > 0 ? "+" : ""}{line}
              </div>
              {coverProb && (
                <div style={{
                  color: edgeColor(spreadEdge),
                  fontSize: "13px",
                  fontWeight: "600"
                }}>
                  {coverProb}% cover
                  {spreadEdge !== null && (
                    <span style={{
                      fontSize: "11px",
                      marginLeft: "4px"
                    }}>
                      {spreadEdge > 0 ? "+" : ""}
                      {spreadEdge}pp edge
                    </span>
                  )}
                </div>
              )}
            </div>
          </div>
        </div>
      )
    }
    
    // O/U row
    return (
      <div style={{
        borderBottom: "1px solid #1f1f1f",
        background: (overEdge >= 5 || underEdge >= 5)
          ? "rgba(0,212,170,0.04)" : "#111111"
      }}>
        <div style={{
          padding: "10px 20px 4px",
          color: "#71717a",
          fontSize: "11px",
          textTransform: "uppercase",
          letterSpacing: "0.05em",
          display: "flex",
          alignItems: "center",
          gap: "6px"
        }}>
          {label}
          {isEstimated && (
            <span style={{
              color: "#f59e0b",
              fontSize: "10px"
            }}>~est</span>
          )}
        </div>
        <div style={{
          display: "grid",
          gridTemplateColumns: "1fr 1fr",
          gap: "1px",
          padding: "0 20px 12px"
        }}>
          {/* Over */}
          <div style={{
            background: overEdge >= 5
              ? "rgba(0,212,170,0.08)"
              : "rgba(255,255,255,0.03)",
            borderRadius: "8px",
            padding: "10px 12px",
            border: overEdge >= 5
              ? "1px solid rgba(0,212,170,0.2)"
              : "1px solid #27272a"
          }}>

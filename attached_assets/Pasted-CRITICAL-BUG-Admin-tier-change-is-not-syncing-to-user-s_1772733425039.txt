ADD normalizeTier to a shared utility file

  // lib/tierUtils.ts
  export const normalizeTier = (raw?: string | null) => {
    if (!raw) return "free"
    const t = raw.toString().toLowerCase().trim()
    // Match any pro variant
    if (t.includes("all_sports") || 
        t.includes("all sports") ||
        t.includes("pro_all") ||
        t === "all") return "pro_all"
    if (t.includes("nba") || 
        t.includes("pro") || 
        t.includes("25") ||
        t.includes("40") ||
        t.includes("49") ||
        t.includes("50") ||
        t === "1" ||
        t === "subscriber") return "pro_nba"
    return "free"
  }
  
  export const hasProAccess = (raw?: string | null) => {
    const t = normalizeTier(raw)
    return t === "pro_nba" || t === "pro_all"
  }

Import hasProAccess in the tab rendering component.
Replace ALL inline tier checks with hasProAccess().

  export async function GET(req) {

    if (!userId) return Response.json(
      { error: "Unauthorized" }, { status: 401 }
    )
    
    const user = await getUser(userId)
    const rawTier = user?.tier ?? user?.plan ?? "free"
    const tier = normalizeTier(rawTier)

      hasProAccess: hasProAccess(rawTier),

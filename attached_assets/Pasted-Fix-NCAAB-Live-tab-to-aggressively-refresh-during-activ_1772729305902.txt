STEP 2 — IMPLEMENT TIERED AUTO-REFRESH

Refresh rate must adapt to what is currently happening
on the slate. More action = faster refresh.

  const getRefreshInterval = (games) => {
    const hasLive = games.some(g => 
      g.status === "in_progress" || g.status === "live"
    )
    const hasHalftime = games.some(g => 
      g.status === "halftime"
    )
    const hasUpcoming = games.some(g => {
      const tipoff = new Date(g.date).getTime()
      const minsUntil = (tipoff - Date.now()) / 60000
      return minsUntil <= 30 && minsUntil > 0
    })
    
    if (hasLive) return 20000        // 20s — games live
    if (hasHalftime) return 15000    // 15s — halftime active
    if (hasUpcoming) return 60000    // 1min — games imminent
    return 300000                    // 5min — idle slate
  }

  const refreshTimeoutRef = useRef(null)
  
  const scheduleNextRefresh = (games) => {
    if (refreshTimeoutRef.current) {

STEP 3 — FIX "NO LIVE GAMES" FALSE POSITIVE

The empty state currently shows when games array
is empty OR when status fields don't match filters.

Add status normalization before checking:

  const normalizeGameStatus = (game) => {
    const statusType = game.status?.type?.name ?? ""
    const statusDesc = game.status?.type?.description
      ?.toLowerCase() ?? ""
    const clock = game.status?.displayClock ?? ""
    const period = game.status?.period ?? 0
    
    // Normalize to standard values
    if (
      statusType === "STATUS_IN_PROGRESS" ||
      statusDesc.includes("in progress") ||
      statusDesc.includes("half") ||
      (period > 0 && clock !== "0:00" && 
       !statusDesc.includes("final"))
    ) {
      if (
        statusDesc.includes("half") ||
        statusDesc.includes("halftime")
      ) return "halftime"
      return "in_progress"
    }
    if (
      statusType === "STATUS_FINAL" ||
      statusDesc.includes("final")
    ) return "final"
    if (
      statusType === "STATUS_SCHEDULED" ||
      statusDesc.includes("scheduled")
    ) return "scheduled"
    
    return "scheduled"
  }

Apply normalizeGameStatus to every game object

  const isMarchMadness = (games) => {
    return games.some(g => 
      g.season?.slug === "post-season" ||
      g.season?.type === 3 ||
      g.competitions?.[0]?.notes?.some(n =>
        n.headline?.toLowerCase().includes("ncaa") ||
        n.headline?.toLowerCase().includes("tournament") ||
        n.headline?.toLowerCase().includes("march madness")
      )
    )
  }
  
  // Override interval for tournament games
  const getTournamentAwareInterval = (games) => {
    if (isMarchMadness(games)) {
      const hasLive = games.some(g =>
        g.status === "in_progress" || 
        g.status === "halftime"

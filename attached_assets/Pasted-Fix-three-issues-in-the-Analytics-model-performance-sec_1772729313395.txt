STEP 3 — AUTO-SETTLE FROM NEXT-DAY STAT FETCH

When the NBA scoreboard refreshes with a new slate,
detect that yesterday's games are now final and
pull box score stats to settle all pending plays.

TRIGGER: When fetchNBAGames() returns a new set of
games where none of yesterday's gameIds appear,
it means yesterday's slate is complete.

  const settlePreviousSlate = async (
    currentGames, pendingPlays
  ) => {
    if (pendingPlays.length === 0) return
    
    // Get unique gameIds from pending plays
    const pendingGameIds = [
      ...new Set(pendingPlays.map(p => p.gameId))
    ]
    
    // Check if any pending games are in current slate
    const currentGameIds = new Set(
      currentGames.map(g => g.id)
    )
    
    // Games that are no longer in current slate
    // = yesterday's games that are now final
    const settleableGameIds = pendingGameIds.filter(
      id => !currentGameIds.has(id)
    )
    
    if (settleableGameIds.length === 0) return
    
    // Fetch box score for each completed game
    for (const gameId of settleableGameIds) {
      await settleGamePlays(gameId, pendingPlays)
    }
  }

SETTLE EACH GAME:

  const settleGamePlays = async (gameId, pendingPlays) => {
    // Fetch final box score from ESPN
    const boxScoreUrl = 
      `https://site.api.espn.com/apis/site/v2/` +
      `sports/basketball/nba/summary?event=${gameId}`
    
    const res = await fetch(boxScoreUrl)
    const data = await res.json()
    
    // Build player stats map from box score
    const playerStats = {}
    
    const players = data.boxscore?.players ?? []
    players.forEach(teamPlayers => {
      teamPlayers.statistics?.forEach(statGroup => {
        statGroup.athletes?.forEach(athlete => {
          const id = athlete.athlete?.id
          if (!id) return
          
          // Map stat abbreviations to values
          const stats = {}
          statGroup.keys?.forEach((key, i) => {
            stats[key] = parseFloat(
              athlete.stats?.[i] ?? 0
            )
          })
          
          // Build combined stat fields
          playerStats[id] = {
            PTS: stats.PTS ?? 0,
            REB: stats.REB ?? 0,
            AST: stats.AST ?? 0,
            BLK: stats.BLK ?? 0,
            STL: stats.STL ?? 0,
            "P+R": (stats.PTS ?? 0) + (stats.REB ?? 0),
            "P+A": (stats.PTS ?? 0) + (stats.AST ?? 0),
            "R+A": (stats.REB ?? 0) + (stats.AST ?? 0),
            "P+R+A": (stats.PTS ?? 0) + 
                     (stats.REB ?? 0) + 
                     (stats.AST ?? 0),
            "S+B": (stats.STL ?? 0) + (stats.BLK ?? 0)
          }
        })
      })
    })
    
    // Settle each pending play for this game
    const playsToSettle = pendingPlays.filter(
      p => p.gameId === gameId && p.result === null
    )
    
    for (const play of playsToSettle) {
      const playerFinalStats = playerStats[play.playerId]
      if (!playerFinalStats) continue
      
      const finalStat = playerFinalStats[play.market]
      if (finalStat === undefined) continue
      
      // Determine result
      let result = "MISS"
      if (finalStat === play.line) {
        result = "PUSH"
      } else if (
        play.direction === "Over" && 
        finalStat > play.line
      ) {
        result = "HIT"
      } else if (
        play.direction === "Under" && 
        finalStat < play.line
      ) {
        result = "HIT"
      }
      
      // Update play record
      await updatePlay(play.id, {
        result,
        finalStat,
        settledAt: new Date().toISOString()
      })
    }
  }

    setIsSettling(true)
    
    try {
      // Fetch current NBA slate
      const currentGames = await fetchNBAGames()
      
      // Get all pending plays (not just today)
      const allPending = plays.filter(
        p => p.result === null
      )
      
      if (allPending.length === 0) {
        showToast("No pending plays to settle", "zinc")
        return
      }
      
      // Attempt to settle all pending plays
      // regardless of whether games are in current slate
      const pendingGameIds = [
        ...new Set(allPending.map(p => p.gameId))
      ]
      
      let settledCount = 0
      let failedCount = 0
      
      for (const gameId of pendingGameIds) {
        try {
          await settleGamePlays(gameId, allPending)
          settledCount++
        } catch (err) {
          failedCount++
          console.warn(
            `[SETTLE] Failed for game ${gameId}:`, err
          )
        }
      }
      
      // Refresh plays after settlement
      const updated = await fetchPlays()
      setPlays(updated)
      
      // Show result toast
      if (settledCount > 0 && failedCount === 0) {
        showToast(
          `✓ Settled ${settledCount} game(s)`, 
          "teal"
        )
      } else if (settledCount > 0) {
        showToast(
          `↻ ${settledCount} settled, ` +
          `${failedCount} unavailable`, 
          "amber"
        )
      } else {
        showToast(
          "Box scores not yet available — try again later",
          "zinc"
        )
      }
      
    } finally {
      setIsSettling(false)
    }
  }

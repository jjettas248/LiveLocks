STEP 1 — Add getSeasonPhase() helper in server/storage.ts

Place above calculateProbability():

function getSeasonPhase(gameDate?: string | Date): "early" | "mid" | "late" | "playoffs" {
  const d = gameDate ? new Date(gameDate) : new Date();
  const month = d.getMonth() + 1;

  if (month >= 10 || month <= 12) return "early"; // Oct–Dec
  if (month >= 1 && month <= 2) return "mid";     // Jan–Feb
  if (month >= 3 && month <= 4) return "late";    // Mar–Apr
  return "playoffs";
}

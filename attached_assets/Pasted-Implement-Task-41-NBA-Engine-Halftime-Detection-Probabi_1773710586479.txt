function parseClockToSeconds(clock: string | null | undefined): number {
  if (!clock) return 0;

  if (clock.includes(":")) {
    const [m, s] = clock.split(":").map(Number);
    return m * 60 + s;
  }

  if (clock.includes(".")) {
    return Math.floor(parseFloat(clock));
  }

  return Number(clock) || 0;
}

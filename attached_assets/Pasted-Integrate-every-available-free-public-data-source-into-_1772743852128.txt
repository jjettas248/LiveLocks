Integrate every available free public data source 
into the NCAAB engine. No new API keys required.
All endpoints are publicly accessible from Replit.
If any endpoint gets blocked, fail silently and 
continue with remaining sources.

Build a unified data pipeline that runs in parallel
and feeds a composite engine score.

---

ARCHITECTURE OVERVIEW

Every NCAAB game card pulls from this source stack:

  Layer 1 — Lines & Markets:
    ESPN pickcenter (free)
    The Odds API (ODDS_API_KEY already in env)
    VegasInsider public scrape
    ActionNetwork public API
    PrizePicks public API
    Underdog public API

  Layer 2 — Advanced Team Analytics:
    BartTorvik / T-Rank (free, no key)
    TeamRankings (free public)
    Sports Reference CBB (free public)
    ESPN team stats + standings

  Layer 3 — Context & Situational:
    ESPN game predictor
    ESPN H2H schedule
    ActionNetwork public betting %
    Rotowire public injury feed

  Layer 4 — Engine:
    Composite score from all available layers
    Each source weighted by reliability
    Missing sources gracefully excluded

      })

      }
      
      const data = await res.json()

        // Adjusted efficiency per 100 possessions
        adjO: parseFloat(match.adjoe ?? match.AdjOE ?? 0),
        adjD: parseFloat(match.adjde ?? match.AdjDE ?? 0),

  const fetchActionNetwork = async (
    homeTeam, awayTeam
  ) => {
    try {
      // ActionNetwork NCAAB scoreboard
      const res = await fetch(
        `https://api.actionnetwork.com/web/v1/` +
        `scoreboard/ncaab?` +
        `period=game&bookIds=15,30,76,123,69,68`,
        {
          headers: {
            "User-Agent": "Mozilla/5.0",
            "Referer": "https://www.actionnetwork.com"
          }
        }
      )
      if (!res.ok) {
        console.warn("[AN] Status:", res.status)
        return null
      }
      
      const data = await res.json()
      const games = data.games ?? []
      
      // Find matching game
      const match = games.find(g => {
        const teams = g.teams ?? []
        const names = teams.map(
          t => t.display_name?.toLowerCase() ?? ""
        )
        const home = homeTeam.toLowerCase()
        const away = awayTeam.toLowerCase()
        
        return names.some(n => 
          n.includes(home.split(" ").pop()) ||
          home.includes(n.split(" ").pop())
        ) && names.some(n =>
          n.includes(away.split(" ").pop()) ||
          away.includes(n.split(" ").pop())
        )
      })
      
      if (!match) {
        console.warn("[AN] No game match")
        return null
      }
      
      // Extract betting percentages
      const odds = match.odds ?? []
      const consensus = odds.find(
        o => o.book_id === 0  // 0 = consensus
      ) ?? odds[0]
      
      console.log("[AN] Found game:", 
        match.teams?.map(t => t.display_name)
      )
      
      return {
        overPct: consensus?.over_bets_pct ?? null,
        underPct: consensus?.under_bets_pct ?? null,
        overMoney: consensus?.over_money_pct ?? null,
        underMoney: consensus?.under_money_pct ?? null,
        total: consensus?.total ?? null,
        spread: consensus?.spread ?? null,
        homeSpreadPct: consensus?.home_spread_bets_pct 
          ?? null,
        awaySpreadPct: consensus?.away_spread_bets_pct 
          ?? null,
        // Line movement
        openTotal: consensus?.open_total ?? null,
        openSpread: consensus?.open_spread ?? null,
        source: "action_network"
      }
      
    } catch (err) {
      console.warn("[AN] Error:", err.message)
      return null
    }
  }

  const fetchVegasInsider = async (

        }

      if (!res.ok) return null
      const html = await res.text()
      
      // Find game section matching team names
      const homeLast = homeTeam.split(" ").pop()
      const awayLast = awayTeam.split(" ").pop()
      
      // VI HTML contains opening + current lines
      // Pattern: team name near odds numbers
      const gameSection = extractVIGameSection(
        html, homeLast, awayLast
      )
      
      if (!gameSection) return null
      
      // Extract opening line and current line
      const openLine = extractNumber(
        gameSection, "open"
      )
      const currentLine = extractNumber(
        gameSection, "current"
      )
      
      return {
        openTotal: openLine,
        currentTotal: currentLine,
        movement: openLine && currentLine
          ? parseFloat(
              (currentLine - openLine).toFixed(1)
            )
          : null,
        source: "vegasinsider"
      }
      
    } catch (err) {
      console.warn("[VI] Error:", err.message)
      return null
    }

        : "Healthy"

  const buildCompositeEngine = (enriched, game) => {
    const line = enriched.odds?.overUnder
    if (!line) return buildNCAABEngine(
      game, enriched.odds, enriched.espnSummary
    )
    
    const signals = []
    let projTotal = null
    let confidence = 0
    
    // SIGNAL 1: BartTorvik tempo projection
    // Most reliable signal when available
    const ht = enriched.torvik.home
    const at = enriched.torvik.away
    
    if (ht?.tempo && at?.tempo && 
        ht?.adjO && at?.adjD &&
        at?.adjO && ht?.adjD) {
      
      // Possessions = avg tempo of both teams
      const possessions = (ht.tempo + at.tempo) / 2
      
      // Expected points per possession
      // AdjO and AdjD are per-100-possession rates
      const homeExpPts = 
        (ht.adjO + at.adjD) / 2 / 100 * possessions
      const awayExpPts = 
        (at.adjO + ht.adjD) / 2 / 100 * possessions
      
      const torvikTotal = homeExpPts + awayExpPts
      
      console.log("[COMPOSITE] Torvik projection:",
        homeExpPts.toFixed(1), "+",
        awayExpPts.toFixed(1), "=",
        torvikTotal.toFixed(1)
      )
      
      signals.push({
        name: "BartTorvik Efficiency",
        projTotal: torvikTotal,
        weight: 4.0,  // highest weight
        diff: torvikTotal - line
      })
      projTotal = torvikTotal
    }
    
    // SIGNAL 2: ESPN team stats scoring environment
    const homeStats = enriched.espnStats.home
    const awayStats = enriched.espnStats.away
    
    if (homeStats?.ppg && awayStats?.ppg &&
        homeStats?.oppPpg && awayStats?.oppPpg) {
      
      const espnTotal = 
        ((homeStats.ppg + awayStats.oppPpg) / 2) +
        ((awayStats.ppg + homeStats.oppPpg) / 2)
      
      signals.push({
        name: "ESPN Pace Model",
        projTotal: espnTotal,
        weight: 2.0,
        diff: espnTotal - line
      })
      
      if (!projTotal) projTotal = espnTotal
    }
    
    // SIGNAL 3: ESPN predictor
    const pred = enriched.espnSummary?.predictor
    const homeProj = parseFloat(
      pred?.homeTeam?.gameProjection ?? 0
    )
    const awayProj = parseFloat(
      pred?.awayTeam?.gameProjection ?? 0
    )
    
    if (homeProj > 20 && awayProj > 20) {
      const espnPredTotal = homeProj + awayProj
      signals.push({
        name: "ESPN Predictor",
        projTotal: espnPredTotal,
        weight: 3.0,
        diff: espnPredTotal - line
      })
      if (!projTotal) projTotal = espnPredTotal
    }
    
    // SIGNAL 4: Line movement direction
    // Sharp money indicator
    const anData = enriched.actionNetwork
    const viData = enriched.vegasInsider
    
    let lineMovementSignal = 0
    
    if (anData?.openTotal && anData?.total) {
      const move = anData.total - anData.openTotal
      // Line moved up = sharp over action
      // Line moved down = sharp under action
      lineMovementSignal = move * 0.5
      signals.push({
        name: "Line Movement",
        projTotal: null,
        weight: 1.5,
        diff: lineMovementSignal
      })
    }
    
    // SIGNAL 5: Public betting % fade signal
    // If public is >70% one side, fade slightly
    if (anData?.overPct !== null && 
        anData?.underPct !== null) {
      const publicBias = anData.overPct - 50
      // Public heavy on over = slight under lean
      const fadeBias = -publicBias * 0.1
      signals.push({
        name: "Public Fade",
        projTotal: null,
        weight: 0.5,
        diff: fadeBias
      })
    }
    
    // SIGNAL 6: Injury adjustment
    const homeInjury = enriched.injuries.home
    const awayInjury = enriched.injuries.away
    
    let injuryAdj = 0
    if (homeInjury?.out > 0) injuryAdj -= 2.5
    if (awayInjury?.out > 0) injuryAdj -= 2.5
    
    if (injuryAdj !== 0) {
      signals.push({
        name: "Injury Reduction",
        projTotal: null,
        weight: 2.0,
        diff: injuryAdj
      })
    }
    
    // COMPOSITE CALCULATION
    // Weighted average of all diff signals
    if (signals.length === 0) {
      return buildNCAABEngine(
        game, enriched.odds, enriched.espnSummary
      )
    }
    
    const totalWeight = signals.reduce(
      (sum, s) => sum + s.weight, 0
    )
    const weightedDiff = signals.reduce(
      (sum, s) => sum + (s.diff * s.weight), 0
    ) / totalWeight
    
    // Final projected total
    const finalProjTotal = projTotal
      ? projTotal + (weightedDiff * 0.3)
      : line + weightedDiff
    
    // Convert to probability
    // More sources = higher certainty multiplier
    const certaintyMultiplier = Math.min(
      2.5 + (signals.length * 0.5), 5.0
    )
    
    const rawProb = 50 + (weightedDiff * certaintyMultiplier)
    
    // Progressive limits based on source count
    const maxEdge = Math.min(
      30 + (signals.length * 5), 70
    )
    const overProb = parseFloat(Math.min(
      Math.max(rawProb, 100 - maxEdge), 
      maxEdge
    ).toFixed(1))
    
    console.log("[COMPOSITE] Final engine:", {
      signals: signals.length,
      weightedDiff: weightedDiff.toFixed(2),
      finalProjTotal: finalProjTotal.toFixed(1),
      overProb,
      sources: signals.map(s => s.name)
    })
    
    return {
      overProb,
      underProb: parseFloat(
        (100 - overProb).toFixed(1)
      ),
      projTotal: parseFloat(
        finalProjTotal.toFixed(1)
      ),
      signals,
      sourceCount: signals.length,
      source: "composite",
      sourceSummary: signals.map(s => s.name)
        .join(", ")
    }
  }

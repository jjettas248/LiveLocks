Integrate every available free public data source 
into the NCAAB engine. No new API keys required.
All endpoints are publicly accessible from Replit.
If any endpoint gets blocked, fail silently and 
continue with remaining sources.

Build a unified data pipeline that runs in parallel
and feeds a composite engine score.

---

ARCHITECTURE OVERVIEW

Every NCAAB game card pulls from this source stack:

  Layer 1 — Lines & Markets:
    ESPN pickcenter (free)
    The Odds API (ODDS_API_KEY already in env)
    VegasInsider public scrape
    ActionNetwork public API
    PrizePicks public API
    Underdog public API

  Layer 2 — Advanced Team Analytics:
    BartTorvik / T-Rank (free, no key)
    TeamRankings (free public)
    Sports Reference CBB (free public)
    ESPN team stats + standings

  Layer 3 — Context & Situational:
    ESPN game predictor
    ESPN H2H schedule
    ActionNetwork public betting %
    Rotowire public injury feed

  Layer 4 — Engine:
    Composite score from all available layers
    Each source weighted by reliability
    Missing sources gracefully excluded

STEP 1 — BARTTORVIK / T-RANK INTEGRATION

Best free NCAAB analytics source. No key needed.

  const fetchBartTorvik = async (teamName) => {
    // BartTorvik team stats endpoint
    // Returns: AdjO, AdjD, Tempo, EFG%, TOV%, ORB%
    
    const year = new Date().getFullYear()
    const url = 
      `https://barttorvik.com/trank.php` +
      `?json=1&year=${year}`
    
    try {
      const res = await fetch(url, {
        headers: { 
          "User-Agent": "Mozilla/5.0",
          "Accept": "application/json"
        }
      })
      if (!res.ok) {
        console.warn("[TORVIK] Fetch failed:", res.status)
        return null
      }
      
      const data = await res.json()
      // data is array of team objects
      // Find team by name matching
      
      const match = findTeamInArray(data, teamName, [
        "team", "teamname", "TeamName", "name"
      ])
      
      if (!match) {
        console.warn("[TORVIK] No match for:", teamName)
        return null
      }
      
      console.log("[TORVIK] Found:", teamName, match)
      
      return {
        // Adjusted efficiency per 100 possessions
        adjO: parseFloat(match.adjoe ?? match.AdjOE ?? 0),
        adjD: parseFloat(match.adjde ?? match.AdjDE ?? 0),
        // Tempo: possessions per 40 min
        tempo: parseFloat(match.tempo ?? match.Tempo ?? 0),
        // Four factors
        efgPct: parseFloat(match.efg ?? match.EFG ?? 0),
        tovPct: parseFloat(match.tov ?? match.TOV ?? 0),
        orbPct: parseFloat(match.orb ?? match.ORB ?? 0),
        ftRate: parseFloat(match.ftr ?? match.FTR ?? 0),
        // Overall rating
        barthag: parseFloat(
          match.barthag ?? match.Barthag ?? 0
        ),
        // Rank
        rank: parseInt(match.rk ?? match.Rank ?? 999),
        source: "barttorvik"
      }
    } catch (err) {
      console.warn("[TORVIK] Error:", err.message)
      return null
    }
  }

  // Also fetch full team page for more detail
  const fetchBartTorvikTeamPage = async (teamName) => {
    // URL encode team name
    const encoded = encodeURIComponent(teamName)
    const year = new Date().getFullYear()
    const url = 
      `https://barttorvik.com/team-drill.php` +
      `?team=${encoded}&year=${year}&json=1`
    
    try {
      const res = await fetch(url, {
        headers: { "User-Agent": "Mozilla/5.0" }
      })
      if (!res.ok) return null
      const data = await res.json()
      
      return {
        homeAdjO: parseFloat(data.homeAdjOE ?? 0),
        homeAdjD: parseFloat(data.homeAdjDE ?? 0),
        awayAdjO: parseFloat(data.awayAdjOE ?? 0),
        awayAdjD: parseFloat(data.awayAdjDE ?? 0),
        last10: data.last10 ?? null,
        source: "barttorvik_detail"
      }
    } catch (err) {
      console.warn("[TORVIK DETAIL] Error:", err.message)
      return null
    }
  }

PARSE THE ODDS API RESPONSE:

  const parseOddsAPIResponse = (data) => {
    if (!data) return null
    
    // data is either a single event or array item
    const event = Array.isArray(data) ? data[0] : data
    if (!event?.bookmakers?.length) return null
    
    // Aggregate across all bookmakers
    const totalsLines = []
    const spreadsLines = []
    const h2hLines = { home: [], away: [] }
    
    event.bookmakers.forEach(book => {
      book.markets?.forEach(market => {
        
        if (market.key === "totals") {
          const over = market.outcomes?.find(
            o => o.name === "Over"
          )
          if (over?.point) totalsLines.push({
            book: book.title,
            line: over.point,
            overPrice: over.price,
            underPrice: market.outcomes?.find(
              o => o.name === "Under"
            )?.price
          })
        }
        
        if (market.key === "spreads") {
          const home = market.outcomes?.find(o =>
            o.name?.toLowerCase() === 
              event.home_team?.toLowerCase()
          )
          if (home?.point !== undefined) 
            spreadsLines.push({
              book: book.title,
              homeSpread: home.point,
              homePrice: home.price,
              awaySpread: -(home.point),
              awayPrice: market.outcomes?.find(o =>
                o.name !== home.name
              )?.price
            })
        }
        
        if (market.key === "h2h") {
          const home = market.outcomes?.find(o =>
            o.name?.toLowerCase() === 
              event.home_team?.toLowerCase()
          )
          const away = market.outcomes?.find(o =>
            o.name?.toLowerCase() !== 
              event.home_team?.toLowerCase()
          )
          if (home?.price) h2hLines.home.push(home.price)
          if (away?.price) h2hLines.away.push(away.price)
        }
      })
    })
    
    // Use consensus (median of available lines)
    const median = (arr) => {
      if (!arr.length) return null
      const sorted = [...arr].sort((a, b) => a - b)
      return sorted[Math.floor(sorted.length / 2)]
    }
    
    const consensusTotal = median(
      totalsLines.map(t => t.line)
    )
    const consensusSpread = median(
      spreadsLines.map(s => s.homeSpread)
    )
    
    // Convert moneyline to implied win%
    const mlToImplied = (ml) => {
      if (!ml) return null
      return ml < 0
        ? Math.abs(ml) / (Math.abs(ml) + 100) * 100
        : 100 / (ml + 100) * 100
    }
    
    const homeMLMedian = median(h2hLines.home)
    const awayMLMedian = median(h2hLines.away)
    
    return {
      overUnder: consensusTotal,
      spread: consensusSpread,
      spreadDetails: consensusSpread !== null
        ? consensusSpread > 0
          ? `+${consensusSpread}`
          : String(consensusSpread)
        : null,
      homeMoneyline: homeMLMedian,
      awayMoneyline: awayMLMedian,
      homeWinPct: homeMLMedian 
        ? parseFloat(mlToImplied(homeMLMedian)
            .toFixed(1)) 
        : null,
      awayWinPct: awayMLMedian
        ? parseFloat(mlToImplied(awayMLMedian)
            .toFixed(1))
        : null,
      books: totalsLines.slice(0, 3),
      bookCount: event.bookmakers.length,
      source: "odds_api"
    }
  }

BUG 3 — FIX NCAAB PACE-BASED ENGINE FALLBACK

When predictor has no projection scores,
fall back to NCAAB pace model.

NCAAB average scoring pace: 70 points per team
per 40 minutes (D1 average ~140 combined).

  const ncaabPaceEngine = (game, oddsLine) => {
    // Use actual scoring pace if game in progress
    const score = game.competitions?.[0]
    const homeScore = parseInt(
      score?.competitors?.find(
        c => c.homeAway === "home"
      )?.score ?? 0
    )
    const awayScore = parseInt(
      score?.competitors?.find(
        c => c.homeAway === "away"
      )?.score ?? 0
    )
    
    const period = game.status?.period ?? 0
    const clock = game.status?.displayClock ?? "20:00"
    const [mins, secs] = clock.split(":").map(Number)
    const clockRemaining = mins + (secs / 60)
    
    // NCAAB: 2 halves × 20 min each = 40 min total
    const totalMinutes = 40
    let elapsedMinutes = 0
    
    if (period === 1) {
      elapsedMinutes = 20 - clockRemaining
    } else if (period >= 2) {
      elapsedMinutes = 20 + (20 - clockRemaining)
    }
    // OT: add 5 min per OT period
    if (period > 2) {
      elapsedMinutes = 40 + ((period - 2) * 5)
    }
    
    const currentTotal = homeScore + awayScore
    const gameProgress = Math.min(
      elapsedMinutes / totalMinutes, 1.0
    )
    
    if (gameProgress < 0.05) {
      // Pre-game: use odds line as baseline
      // NCAAB D1 average: 140 combined
      const baseline = oddsLine ?? 140
      return {
        projTotal: baseline,
        overProb: null,  // no edge pre-game without proj
        gameProgress: 0
      }
    }
    
    // Project final total from current pace
    const projectedPace = currentTotal / gameProgress
    
    // Regression to mean: 60% pace + 40% league avg
    const leagueAvg = 140
    const projTotal = (projectedPace * 0.60) + 
                      (leagueAvg * 0.40)
    
    if (!oddsLine) {
      return { projTotal, overProb: null, gameProgress }
    }
    
    const diff = projTotal - oddsLine
    
    // Dynamic multiplier based on game progress
    // (reuses existing getDynamicMultiplier logic)
    const multiplier = getDynamicMultiplier({
      gameProgress,
      sport: "ncaab"
    })
    
    const rawProb = 50 + (diff * multiplier)
    const { min, max } = getProgressiveLimits({
      gameProgress
    })
    const overProb = Math.min(
      Math.max(rawProb, min), max
    )
    
    return {
      projTotal: parseFloat(projTotal.toFixed(1)),
      overProb: parseFloat(overProb.toFixed(1)),
      underProb: parseFloat(
        (100 - overProb).toFixed(1)
      ),
      gameProgress
    }
  }

  const handleNCAABCardExpand = async (gameId) => {
    setExpandedGameId(gameId)
    
    const game = ncaabGames.find(g => g.id === gameId)
    if (!game) return
    
    // Fetch in parallel — don't wait for one to finish
    const [summary, oddsData] = await Promise.all([
      game.summary 
        ? Promise.resolve(game.summary)
        : fetchNCAABGameSummary(gameId),
      game.odds
        ? Promise.resolve(game.odds)
        : fetchNCAABOdds(gameId)
    ])
    
    const projections = extractNCAABProjections(summary)
    
    // If ESPN predictor has projections, use them
    // Otherwise fall back to pace model
    let engine
    if (projections.homeProj > 0 && 
        projections.awayProj > 0) {
      const projTotal = 
        projections.homeProj + projections.awayProj
      engine = {
        projTotal,
        overProb: preGameEngineProb(
          projTotal, oddsData?.overUnder
        ),
        source: "espn_predictor"
      }
    } else {
      engine = ncaabPaceEngine(
        game, oddsData?.overUnder
      )
      engine.source = "pace_model"
    }
    
    console.log("[NCAAB] Engine result:", engine)
    console.log("[NCAAB] Odds loaded:", {
      total: oddsData?.overUnder,
      spread: oddsData?.spreadDetails,
      books: oddsData?.bookCount
    })
    
    // Update game with all fetched data
    setNcaabGames(prev => prev.map(g =>
      g.id === gameId 
        ? { ...g, summary, odds: oddsData, engine }
        : g
    ))

export function MLBScheduleList({ games }) {
  if (!games?.length) return null;

  return (
    <div>
      <h2>Today's Games</h2>
      {games.map(game => (
        <div key={game.id}>
          {game.awayTeam} @ {game.homeTeam}
          <div>{formatTime(game.startTime)}</div>
        </div>
      ))}
    </div>
  );
}

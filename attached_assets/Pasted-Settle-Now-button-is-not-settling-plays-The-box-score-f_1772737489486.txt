STEP 2 — FIX ESPN BOX SCORE FETCH

The correct ESPN NBA box score endpoint:

  const fetchBoxScore = async (gameId) => {
    
    // Clean gameId — remove any prefix
    const cleanId = String(gameId)
      .replace(/^(nba-|ncaab-|game-)/i, "")
    
    const url = 
      `https://site.api.espn.com/apis/site/v2/` +
      `sports/basketball/nba/summary?event=${cleanId}`
    
    console.log("[SETTLE] Fetching box score:", url)
    
    let data
    try {
      const res = await fetch(url)
      
      if (!res.ok) {
        console.error(
          "[SETTLE] ESPN returned:", res.status, 
          "for game", cleanId
        )
        return null
      }
      
      data = await res.json()
    } catch (err) {
      console.error("[SETTLE] Fetch failed:", err)
      return null
    }
    
    // Confirm box score exists
    if (!data?.boxscore?.players?.length) {
      console.warn(
        "[SETTLE] No box score data for game", 
        cleanId,
        "- game may not be final yet"
      )
      return null
    }
    
    console.log(
      "[SETTLE] Box score found, teams:", 
      data.boxscore.players.length
    )
    
    return data
  }

BUILD PLAYER STATS WITH CORRECT KEYS:

  const buildPlayerStats = (data) => {
    const playerStats = {}
    
    data.boxscore.players.forEach(teamData => {
      // ESPN has multiple stat categories
      // "starters" and "totals" — use all athletes
      teamData.statistics.forEach(statGroup => {
        const keys = statGroup.keys ?? []
        
        statGroup.athletes.forEach(athlete => {
          const id = String(athlete.athlete?.id)
          if (!id || id === "undefined") return
          
          if (!playerStats[id]) {
            playerStats[id] = {
              name: athlete.athlete?.displayName,
              id
            }
          }
          
          // Map each ESPN key to its value
          const rawStats = {}
          keys.forEach((key, index) => {
            const val = athlete.stats?.[index]
            rawStats[key.toUpperCase()] = 
              parseFloat(val ?? 0) || 0
          })
          
          // Store raw ESPN stats
          Object.assign(playerStats[id], rawStats)
        })
      })
    })
    
    return playerStats
  }

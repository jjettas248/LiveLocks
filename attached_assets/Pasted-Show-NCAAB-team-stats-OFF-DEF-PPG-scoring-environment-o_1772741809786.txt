  // Module-level cache — persists across renders
  const teamStatsCache = {}
  const teamRecordCache = {}
  
  const fetchTeamData = async (teamId) => {
    if (teamStatsCache[teamId]) {
      return teamStatsCache[teamId]
    }
    
    // Fetch stats and record in parallel
    const [statsRes, recordRes] = await Promise.all([
      fetch(
        `https://site.api.espn.com/apis/site/v2/` +
        `sports/basketball/mens-college-basketball/` +
        `teams/${teamId}/statistics`
      ),
      fetch(
        `https://site.api.espn.com/apis/site/v2/` +
        `sports/basketball/mens-college-basketball/` +
        `teams/${teamId}`
      )
    ])
    
    const stats = statsRes.ok 
      ? await statsRes.json() : null
    const record = recordRes.ok 
      ? await recordRes.json() : null
    
    // Extract from ESPN stats response
    const getStat = (name) =>
      stats?.splits?.categories
        ?.flatMap(c => c.stats ?? [])
        ?.find(s => s.name === name)
        ?.value ?? null
    
    const result = {
      teamId,
      
      // Scoring
      ppg: getStat("points"),
      oppPpg: getStat("pointsAgainst"),
      
      // Pace indicators
      fgPct: getStat("fieldGoalPct"),
      threePct: getStat("threePointFieldGoalPct"),
      rebPg: getStat("rebounds"),
      astPg: getStat("assists"),
      topg: getStat("turnovers"),
      
      // Record and ranking
      wins: record?.team?.record?.items?.[0]
        ?.summary?.split("-")?.[0] ?? null,
      losses: record?.team?.record?.items?.[0]
        ?.summary?.split("-")?.[1] ?? null,
      apRank: record?.team?.rank ?? null,
      
      // Conference
      conference: record?.team?.groups
        ?.find(g => g.isConference)?.name ?? null
    }
    
    teamStatsCache[teamId] = result
    console.log(
      `[TEAM STATS] Cached ${teamId}:`,
      `${result.ppg} PPG / ${result.oppPpg} OppPPG`
    )
    
    return result
  }

FETCH FOR BOTH TEAMS ON CARD EXPAND:

  const fetchBothTeamStats = async (game) => {
    const competitors = 
      game.competitions?.[0]?.competitors ?? []
    
    const home = competitors.find(
      c => c.homeAway === "home"
    )
    const away = competitors.find(
      c => c.homeAway === "away"
    )
    
    const homeId = home?.team?.id
    const awayId = away?.team?.id
    
    if (!homeId || !awayId) return null
    
    const [homeStats, awayStats] = await Promise.all([
      fetchTeamData(homeId),
      fetchTeamData(awayId)
    ])
    
    // Combined scoring environment
    // Avg of: (home offense + away defense) / 2
    //      + (away offense + home defense) / 2
    const scoringEnv = 
      homeStats?.ppg && awayStats?.ppg &&
      homeStats?.oppPpg && awayStats?.oppPpg
        ? parseFloat((
            ((homeStats.ppg + awayStats.oppPpg) / 2) +
            ((awayStats.ppg + homeStats.oppPpg) / 2)
          ).toFixed(1))
        : null
    
    return { homeStats, awayStats, scoringEnv }
  }

STEP 3 — TEAM STATS DISPLAY COMPONENT

One reusable component used in all three 
card states. Accept cardState prop to 
adjust which rows are shown.

  const TeamStatsRows = ({ 
    homeStats, 
    awayStats, 
    scoringEnv,
    oddsLine,
    homeAbbr,
    awayAbbr,
    cardState  // "scheduled" | "live" | "halftime"
  }) => {
    if (!homeStats && !awayStats) return null
    
    const envVsLine = scoringEnv && oddsLine
      ? scoringEnv - oddsLine
      : null
    
    const envColor = envVsLine === null
      ? "#71717a"
      : envVsLine >= 8 ? "#00d4aa"
      : envVsLine <= -8 ? "#ef4444"
      : "#a1a1aa"
    
    const envLabel = envVsLine === null
      ? "—"
      : envVsLine >= 8 ? "↑ Over lean"
      : envVsLine <= -8 ? "↓ Under lean"
      : "Neutral"
    
    return (
      <>
        {/* HOME OFFENSE / DEFENSE */}
        <StatRow
          label={`${homeAbbr} OFF / DEF`}
          value={
            homeStats?.ppg && homeStats?.oppPpg
              ? `${homeStats.ppg.toFixed(1)} / ` +
                `${homeStats.oppPpg.toFixed(1)}`
              : "—"
          }
          sub="Points per game / Against"
          // Show record + rank badge if ranked
          badge={homeStats?.apRank
            ? `#${homeStats.apRank} AP`
            : null}
          badgeColor="#f59e0b"
        />
        
        {/* AWAY OFFENSE / DEFENSE */}
        <StatRow
          label={`${awayAbbr} OFF / DEF`}
          value={
            awayStats?.ppg && awayStats?.oppPpg
              ? `${awayStats.ppg.toFixed(1)} / ` +
                `${awayStats.oppPpg.toFixed(1)}`
              : "—"
          }
          sub="Points per game / Against"
          badge={awayStats?.apRank
            ? `#${awayStats.apRank} AP`
            : null}
          badgeColor="#f59e0b"
        />
        
        {/* SCORING ENVIRONMENT */}
        <StatRow
          label="SCORING ENV"
          value={scoringEnv?.toFixed(1) ?? "—"}
          valueColor={envColor}
          sub={
            envVsLine !== null
              ? `Env ${scoringEnv} vs Line ` +
                `${oddsLine ?? "—"}`
              : "Combined pace projection"
          }
          subFont="font-mono"
          rightBadge={envLabel}
          rightBadgeColor={envColor}
        />
        
        {/* ADDITIONAL ROWS — live + halftime only */}
        {(cardState === "live" || 
          cardState === "halftime") && (
          <>
            {/* FG% comparison */}
            {homeStats?.fgPct && awayStats?.fgPct && (
              <StatRow
                label="FG% MATCHUP"
                value={
                  `${(homeStats.fgPct * 100)
                    .toFixed(1)}% / ` +
                  `${(awayStats.fgPct * 100)
                    .toFixed(1)}%`
                }
                sub={`${homeAbbr} / ${awayAbbr}`}
              />
            )}
            
            {/* Season record */}
            {homeStats?.wins && awayStats?.wins && (
              <StatRow
                label="RECORD"
                value={
                  `${homeStats.wins}-` +
                  `${homeStats.losses} / ` +
                  `${awayStats.wins}-` +
                  `${awayStats.losses}`
                }
                sub={`${homeAbbr} / ${awayAbbr}`}
              />
            )}
          </>
        )}
      </>
    )
  }

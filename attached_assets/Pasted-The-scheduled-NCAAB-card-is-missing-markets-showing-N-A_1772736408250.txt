BUG 1 — FIX H2H HISTORY SCORES SHOWING N/A

The H2H rows show "–" in place of scores and
"N/A" badges. Scores exist in the data but are
being read from the wrong field.

Fix score extraction in the H2H mapping function:

  const extractH2HGameData = (event, homeTeamId) => {
    const comp = event.competitions?.[0]
    if (!comp) return null
    
    const competitors = comp.competitors ?? []
    
    // Score field is on each competitor object
    // Try multiple field locations
    const getScore = (competitor) => {
      const score = 
        competitor.score ??
        competitor.score?.value ??
        competitor.statistics?.find(
          s => s.name === "points"
        )?.displayValue ??
        null
      
      return score !== null 
        ? parseInt(score) 
        : null
    }
    
    const home = competitors.find(
      c => c.homeAway === "home"
    )
    const away = competitors.find(
      c => c.homeAway === "away"
    )
    
    const homeScore = getScore(home)
    const awayScore = getScore(away)
    
    // Only include if scores exist
    if (homeScore === null || awayScore === null) {
      // Game may not be completed yet
      // Check completed flag
      const isCompleted = 
        comp.status?.type?.completed === true ||
        comp.status?.type?.name === "STATUS_FINAL"
      
      if (!isCompleted) return null // skip non-final
    }
    
    const total = (homeScore ?? 0) + (awayScore ?? 0)
    
    // Get odds from that specific game
    const odds = comp.odds?.[0] ?? null
    const line = odds?.overUnder ?? null
    const spreadDetails = odds?.details ?? null
    
    return {
      id: event.id,
      date: event.date,
      homeTeam: home?.team?.abbreviation ?? "HOME",
      awayTeam: away?.team?.abbreviation ?? "AWAY",
      homeScore,
      awayScore,
      total,
      line,
      spreadDetails,
      overUnderResult: line 
        ? total > line ? "OVER" 
        : total < line ? "UNDER" 
        : "PUSH"
        : null,
      isHomeTeamPlaying: 
        home?.team?.id === homeTeamId ||
        away?.team?.id === homeTeamId
    }
  }

PRE-GAME ENGINE CALC:

  // At tipoff: use projection vs line directly
  // No time-elapsed factor — game hasn't started
  // Use low multiplier (3.0 — minimal certainty)
  
  const preGameEngineProb = (projTotal, line) => {
    if (!projTotal || !line) return null
    const diff = projTotal - line
    const rawProb = 50 + (diff * 3.0)
    // Pre-game: tight limits (30-70 range)
    return parseFloat(
      Math.min(Math.max(rawProb, 30), 70).toFixed(1)
    )
  }
  
  const overProb = preGameEngineProb(projTotal, markets.overUnder)
  const underProb = overProb 
    ? parseFloat((100 - overProb).toFixed(1)) 
    : null

BUG 1 — FIX SPREAD COVER PROBABILITY

Current code is returning 0% for USC covering -2.5
when projections clearly show USC winning by 5.6 pts.

FIND the spread cover probability function.
It is likely using ESPN pickcenter winPercentage
but that field is either null or not being read.

CORRECT SPREAD COVER LOGIC:

  const calculateSpreadCoverProb = (game, summary) => {
    
    // Method 1: Use ESPN pickcenter win percentage
    // This is the most accurate when available
    const espnHomeWinPct = 
      summary?.pickcenter?.[0]
        ?.homeTeamOdds?.winPercentage
    
    if (espnHomeWinPct != null && espnHomeWinPct > 0) {
      return parseFloat(parseFloat(espnHomeWinPct)
        .toFixed(1))
    }
    
    // Method 2: Derive from projections when ESPN 
    // win% is unavailable (fallback)
    const homeProj = parseFloat(
      game.predictor?.homeTeam?.gameProjection ?? 0
    )
    const awayProj = parseFloat(
      game.predictor?.awayTeam?.gameProjection ?? 0
    )
    
    if (!homeProj || !awayProj) return 50
    
    // Get spread value and which team it favors
    // Spread string from ESPN: "USC Trojans -2.5"
    // Negative spread = that team is favored
    const spreadDetails = 
      summary?.pickcenter?.[0]?.details ?? 
      game.odds?.details ?? ""
    
    // Parse spread value
    const spreadMatch = spreadDetails.match(
      /([+-]?\d+\.?\d*)\s*$/
    )
    const spreadValue = spreadMatch 
      ? parseFloat(spreadMatch[1]) 
      : 0
    
    // Determine which competitor has the spread
    // and whether home or away
    const competitors = 
      game.competitions?.[0]?.competitors ?? []
    const homeTeam = competitors
      .find(c => c.homeAway === "home")
    const awayTeam = competitors
      .find(c => c.homeAway === "away")
    
    const spreadFavorsHome = 
      spreadDetails.toLowerCase().includes(
        homeTeam?.team?.displayName?.toLowerCase()
          ?.split(" ")?.[0] ?? "home"
      ) && spreadValue < 0
    
    // Calculate projected margin
    const projMargin = homeProj - awayProj
    // Positive = home winning
    
    // Adjust margin against spread
    // If home favored by -2.5: need to win by >2.5
    const adjustedMargin = spreadFavorsHome
      ? projMargin - Math.abs(spreadValue)  
      // home needs to cover: win by more than spread
      : projMargin + Math.abs(spreadValue)  
      // away needs to cover: keep it within spread
    
    // Convert margin to probability
    // Each projected point of margin = ~6% probability shift
    // Clamped by game progress limits
    const rawProb = 50 + (adjustedMargin * 6)
    const { min, max } = getProgressiveLimits(game)
    
    return parseFloat(
      Math.min(Math.max(rawProb, min), max).toFixed(1)
    )
  }

USC EXAMPLE VERIFICATION:
  homeProj (USC) = 82.7
  awayProj (WASH) = 77.1
  spread = USC -2.5 (USC favored, must win by >2.5)
  projMargin = 82.7 - 77.1 = +5.6 (USC winning)
  adjustedMargin = 5.6 - 2.5 = +3.1 (covers by 3.1)
  rawProb = 50 + (3.1 * 6) = 68.6%
  → USC cover should show ~68-70% not 0%

CORRECT ESPN H2H ENDPOINT:

  // Get team IDs from current game competitors
  const homeTeamId = game.competitions?.[0]
    ?.competitors
    ?.find(c => c.homeAway === "home")
    ?.team?.id

  const awayTeamId = game.competitions?.[0]
    ?.competitors
    ?.find(c => c.homeAway === "away")
    ?.team?.id

  // Fetch home team's schedule to find H2H games
  const scheduleUrl = 
    `https://site.api.espn.com/apis/site/v2/sports/` +
    `basketball/mens-college-basketball/teams/` +
    `${homeTeamId}/schedule`

  const scheduleRes = await fetch(scheduleUrl)
  const scheduleData = await scheduleRes.json()

  // Filter for games where opponent = awayTeamId
  const h2hGames = scheduleData.events
    ?.filter(event => {
      const opponents = event.competitions?.[0]
        ?.competitors ?? []
      return opponents.some(c => 
        c.team?.id === awayTeamId && 
        event.competitions?.[0]?.status
          ?.type?.completed === true
      )
    })
    ?.slice(0, 3) // last 3 H2H games
    ?? []

  // Extract data from each H2H game
  const h2hData = h2hGames.map(event => {
    const comp = event.competitions?.[0]
    const competitors = comp?.competitors ?? []
    
    const home = competitors
      .find(c => c.homeAway === "home")
    const away = competitors
      .find(c => c.homeAway === "away")
    
    const homeScore = parseInt(home?.score ?? 0)
    const awayScore = parseInt(away?.score ?? 0)
    const total = homeScore + awayScore
    
    // Get the line from that game if available
    const line = comp?.odds?.[0]?.overUnder ?? null
    const spreadDetails = 
      comp?.odds?.[0]?.details ?? null
    
    // Determine Over/Under result
    const overUnderResult = line 
      ? total > line 
        ? "OVER" 
        : total < line 
        ? "UNDER" 
        : "PUSH"
      : null
    
    // Determine spread coverage
    const spreadCovered = spreadDetails 
      ? determineSpreadCoverage(
          comp, spreadDetails, homeScore, awayScore
        ) 
      : null
    
    return {
      date: event.date,
      homeTeam: home?.team?.abbreviation,
      awayTeam: away?.team?.abbreviation,
      homeScore,
      awayScore,
      total,
      line,
      overUnderResult,
      spreadDetails,
      spreadCovered,
      location: home?.team?.id === homeTeamId 
        ? "HOME" : "AWAY"
    }
  })

  const determineSpreadCoverage = (
    comp, spreadDetails, homeScore, awayScore
  ) => {
    const spreadMatch = 
      spreadDetails.match(/([+-]?\d+\.?\d*)\s*$/)
    if (!spreadMatch) return null
    
    const spreadValue = parseFloat(spreadMatch[1])
    const margin = homeScore - awayScore
    
    if (spreadValue < 0) {
      // Home team favored
      if (margin > Math.abs(spreadValue)) 
        return "HOME COVERED"
      if (margin === Math.abs(spreadValue)) 
        return "PUSH"
      return "AWAY COVERED"
    } else {
      // Away team favored
      if (awayScore - homeScore > Math.abs(spreadValue))
        return "AWAY COVERED"
      if (awayScore - homeScore === Math.abs(spreadValue))
        return "PUSH"
      return "HOME COVERED"
    }
  }

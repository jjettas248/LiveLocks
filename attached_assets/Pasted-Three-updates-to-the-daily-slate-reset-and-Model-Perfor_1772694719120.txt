STEP CYCLING LOGIC:

  const RESET_STEPS = [
    "Clearing yesterday's slate...",
    "Fetching today's NBA games...",
    "Fetching today's NCAAB games...",
    "Engine ready...",
    "Let's go 🔒"
  ]

  const [resetStep, setResetStep] = useState(0)
  const [showResetOverlay, setShowResetOverlay] = 
    useState(false)

  const executeSlateReset = async () => {
    setShowResetOverlay(true)
    setResetStep(0)
    
    // Cycle through steps
    const stepInterval = setInterval(() => {
      setResetStep(prev => {
        if (prev >= RESET_STEPS.length - 1) {
          clearInterval(stepInterval)
          return prev
        }
        return prev + 1
      })
    }, 600)
    
    // Run actual reset operations in parallel
    await Promise.all([
      clearDailyState(),
      fetchNBAScoreboard(),
      fetchNCAABScoreboard()
    ])
    
    // Wait for animation to complete
    await new Promise(r => setTimeout(r, 2800))
    
    // Fade out overlay
    setTimeout(() => {
      setShowResetOverlay(false)
      setResetStep(0)
    }, 300)
    
    // Save reset timestamp
    localStorage.setItem(
      "lastSlateReset",
      new Date().toISOString()
    )
  }

SAVE RESET TIME:

  Store in DB and localStorage:

    const saveResetTime = async (timeStr) => {
      // timeStr format: "06:00" or "05:30"
      const [hours, minutes] = timeStr
        .split(":").map(Number)
      
      // Save to DB via API
      await fetch("/api/admin/settings", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ 
          slateResetHour: hours,
          slateResetMinute: minutes

      // Save to localStorage as backup

READ RESET TIME ON LOAD:

  const getResetTime = async () => {
    // Try DB first
    try {
      const res = await fetch("/api/admin/settings")
      const data = await res.json()
      if (data.slateResetHour != null) {
        return {
          hours: data.slateResetHour,
          minutes: data.slateResetMinute ?? 0
        }
      }
    } catch (e) {}
    
    // Fallback to localStorage
    try {
      const stored = localStorage.getItem(
        "slateResetTime"
      )
      if (stored) return JSON.parse(stored)
    } catch (e) {}
    
    // Default: 6 AM EST
    return { hours: 6, minutes: 0 }
  }

DYNAMIC RESET SCHEDULING:

  Replace hardcoded 11:00 UTC with dynamic calc:

  const getNextResetTime = async () => {
    const { hours, minutes } = await getResetTime()
    
    // Convert EST to UTC: add 5 hours
    // EST = UTC-5, so 6 AM EST = 11 AM UTC
    const resetHourUTC = hours + 5
    const resetMinuteUTC = minutes
    
    const next = new Date()
    next.setUTCHours(resetHourUTC, resetMinuteUTC, 0, 0)
    
    if (new Date() >= next) {
      next.setUTCDate(next.getUTCDate() + 1)
    }
    
    return next
  }

RESCHEDULE ON CHANGE:

  const rescheduleResetTimer = (hours, minutes) => {
    // Clear existing timer
    if (resetTimerRef.current) {
      clearTimeout(resetTimerRef.current)
    }
    
    // Calculate new delay
    const resetHourUTC = hours + 5
    const next = new Date()
    next.setUTCHours(resetHourUTC, minutes, 0, 0)
    if (new Date() >= next) {
      next.setUTCDate(next.getUTCDate() + 1)
    }
    
    const delay = next.getTime() - Date.now()
    
    resetTimerRef.current = setTimeout(() => {
      executeSlateReset()
      rescheduleResetTimer(hours, minutes)
    }, delay)
    
    console.log(
      `[RESET] Rescheduled: ` +
      `${hours}:${String(minutes).padStart(2,"0")} EST ` +
      `(in ${Math.round(delay/60000)}m)`
    )
  }

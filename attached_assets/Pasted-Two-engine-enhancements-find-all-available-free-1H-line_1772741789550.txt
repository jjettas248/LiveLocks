HELPER — extract 1H from alternate totals:

  const extractH1FromAlternates = (data, fullGameLine) => {
    if (!data?.bookmakers?.length) return null
    
    // Full game NCAAB ~140, H1 ~65 (47%)
    // Look for lines in 55-75 range
    const h1Range = { min: 50, max: 80 }
    
    const candidates = []
    data.bookmakers.forEach(book => {
      book.markets?.forEach(market => {
        if (market.key === "alternate_totals") {
          market.outcomes?.forEach(o => {
            if (o.point >= h1Range.min && 
                o.point <= h1Range.max) {
              candidates.push(o.point)
            }
          })
        }
      })
    })
    
    if (!candidates.length) return null
    
    // Return median candidate
    const sorted = [...candidates].sort((a, b) => a - b)
    return sorted[Math.floor(sorted.length / 2)]
  }

When the scoring environment diverges significantly

UPDATE getPreGameConfidenceTier:

  const getPreGameConfidenceTier = (
    overProb, 
    scoringEnv = null,
    oddsLine = null
  ) => {
    if (!overProb) return null
    
    let edge = Math.abs(overProb - 50)
    
    // SCORING ENVIRONMENT BOOST
    // If scoringEnv diverges from line, engine has
    // additional structural support for the lean
    let envBoost = 0
    let envSignal = null
    
    if (scoringEnv && oddsLine) {
      const envDiff = scoringEnv - oddsLine
      
      if (Math.abs(envDiff) >= 8) {
        // Significant structural divergence
        envBoost = 3  // Add 3pp to edge for tier calc
        
        envSignal = envDiff > 0
          ? "High-pace matchup vs low line"
          : "Low-pace matchup vs high line"
      } else if (Math.abs(envDiff) >= 4) {
        envBoost = 1.5
        envSignal = envDiff > 0
          ? "Pace leans Over"
          : "Pace leans Under"
      }
    }
    
    const adjustedEdge = edge + envBoost
    
    // Tier thresholds (same as before but
    // adjusted edge can push to next tier)
    let tier
    if (adjustedEdge < 3) {
      tier = {
        label: "No Edge",
        sublabel: "Model sees even matchup",
        color: "#71717a",
        bg: "rgba(255,255,255,0.04)",
        border: "rgba(255,255,255,0.08)"
      }
    } else if (adjustedEdge < 7) {
      tier = {
        label: "Low Confidence",
        sublabel: "Slight lean — insufficient for signal",
        color: "#71717a",
        bg: "rgba(255,255,255,0.04)",
        border: "rgba(255,255,255,0.08)"
      }
    } else if (adjustedEdge < 12) {
      tier = {
        label: "Moderate Signal",
        sublabel: "Pre-game lean — monitor at tipoff",
        color: "#f59e0b",
        bg: "rgba(245,158,11,0.08)",
        border: "rgba(245,158,11,0.2)"
      }
    } else if (adjustedEdge < 18) {
      tier = {
        label: "Strong Pre-Game Signal",
        sublabel: "Model has clear lean before tipoff",
        color: "#00d4aa",
        bg: "rgba(0,212,170,0.08)",
        border: "rgba(0,212,170,0.2)"
      }
    } else {
      tier = {
        label: "High Confidence",
        sublabel: "Significant model edge pre-game",
        color: "#00d4aa",
        bg: "rgba(0,212,170,0.12)",
        border: "rgba(0,212,170,0.25)"
      }
    }
    
    // Attach environment signal if boost occurred
    if (envSignal && envBoost > 0) {
      tier.envSignal = envSignal
      tier.envBoost = envBoost
      tier.boosted = true
    }
    
    return tier
  }

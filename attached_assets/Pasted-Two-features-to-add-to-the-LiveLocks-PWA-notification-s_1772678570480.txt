Two features to add to the LiveLocks PWA notification system.

---

1. DEEP-LINK FROM NOTIFICATION TO SPECIFIC CARD

When a user taps a push notification, open the app 
directly to the specific game or player card — 
not the home screen.

---

NOTIFICATION PAYLOAD — ADD DEEP LINK DATA

When constructing every notification, include 
routing data in the payload:

  For NBA prop notifications:
    data: {
      url: "/",
      tab: "nba",
      gameId: play.gameId,
      playerId: play.playerId,
      market: play.market,
      cardType: "prop"
    }

  For NCAAB game notifications:
    data: {
      url: "/",
      tab: "ncaab",
      gameId: game.id,
      cardType: "game"
    }

  For edge-flip / direction-change notifications:
    data: {
      url: "/",
      tab: game.sport === "nba" ? "nba" : "ncaab",
      gameId: game.id,
      cardType: "game",
      trigger: "edge_flip"
    }

---

SERVICE WORKER — HANDLE NOTIFICATION CLICK

In the service worker notificationclick handler:

  self.addEventListener("notificationclick", event => {
    event.notification.close()
    if (event.action === "dismiss") return
    
    const data = event.notification.data || {}
    
    // Build deep link URL with query params
    const params = new URLSearchParams()
    if (data.tab) params.set("tab", data.tab)
    if (data.gameId) params.set("gameId", data.gameId)
    if (data.playerId) params.set("playerId", data.playerId)
    if (data.cardType) params.set("cardType", data.cardType)
    if (data.trigger) params.set("trigger", data.trigger)
    
    const targetUrl = `/?${params.toString()}`
    
    event.waitUntil(
      clients.matchAll({ 
        type: "window", 
        includeUncontrolled: true 
      }).then(clientList => {
        // App already open — focus and navigate
        for (const client of clientList) {
          if ("focus" in client) {
            client.focus()
            client.postMessage({
              type: "NOTIFICATION_NAVIGATE",
              data
            })
            return
          }
        }
        // App not open — open with deep link params
        if (clients.openWindow) {
          return clients.openWindow(targetUrl)
        }
      })
    )
  })

---

APP — HANDLE DEEP LINK ON LOAD AND MESSAGE

In the main app component, handle two scenarios:

SCENARIO A — App was closed, opened via notification:
  On mount, check URL query params:
  
  useEffect(() => {
    const params = new URLSearchParams(window.location.search)
    const tab = params.get("tab")
    const gameId = params.get("gameId")
    const playerId = params.get("playerId")
    const cardType = params.get("cardType")
    
    if (tab) {
      setActiveTab(tab)
    }
    
    if (gameId && cardType === "game") {
      // Small delay to let tab render first
      setTimeout(() => {
        setExpandedGameId(gameId)
        rowRefs.current[gameId]?.scrollIntoView({
          behavior: "smooth",
          block: "start"
        })
      }, 400)
    }
    
    if (playerId && cardType === "prop") {
      setTimeout(() => {
        setExpandedPlayId(playerId)
        playRefs.current[playerId]?.scrollIntoView({
          behavior: "smooth",
          block: "start"
        })
      }, 400)
    }
    
    // Clean URL after reading params
    window.history.replaceState({}, "", "/")
  }, [])

SCENARIO B — App was open, notification received:
  Listen for postMessage from service worker:
  
  useEffect(() => {
    const handleMessage = (event) => {
      if (event.data?.type !== "NOTIFICATION_NAVIGATE") 
        return
      
      const { tab, gameId, playerId, cardType } = 
        event.data.data
      
      if (tab) setActiveTab(tab)
      
      setTimeout(() => {
        if (gameId && cardType === "game") {
          setExpandedGameId(gameId)
          rowRefs.current[gameId]?.scrollIntoView({
            behavior: "smooth",
            block: "start"
          })
        }
        if (playerId && cardType === "prop") {
          setExpandedPlayId(playerId)
          playRefs.current[playerId]?.scrollIntoView({
            behavior: "smooth",
            block: "start"
          })
        }
      }, 300)
    }
    
    navigator.serviceWorker?.addEventListener(
      "message", handleMessage
    )
    return () => navigator.serviceWorker
      ?.removeEventListener("message", handleMessage)
  }, [])

NOTIFICATION LOG ENTRY SHAPE

Store in component state (in-memory, resets daily):

  interface NotificationLog {
    id: string
    timestamp: number          // Date.now()
    type: "prop" | "game" | "edge_flip"
    sport: "nba" | "ncaab"
    title: string              // notification title sent
    body: string               // notification body sent
    confidence: number         // engine prob at time sent
    gameId: string
    playerId?: string
    market?: string
    direction?: string
    line?: number
    result?: "HIT" | "MISS" | "PUSH" | null
    settledAt?: number
  }

Every time a notification is sent, push a log entry:
  
  const logNotification = (play, notifData) => {
    const entry: NotificationLog = {
      id: `notif-${Date.now()}-${play.id}`,
      timestamp: Date.now(),
      type: play.cardType || "prop",
      sport: play.sport || "nba",
      title: notifData.title,
      body: notifData.body,
      confidence: play.engineProb,
      gameId: play.gameId,
      playerId: play.playerId,
      market: play.market,
      direction: play.direction,
      line: play.line,
      result: null  // populated after settlement
    }
    setNotificationLog(prev => [entry, ...prev])
  }

RESULT SETTLEMENT

When a game goes Final, update all log entries 
for that game with HIT/MISS result:

  const settleNotificationLogs = (game) => {
    setNotificationLog(prev => prev.map(entry => {
      if (entry.gameId !== game.id) return entry
      if (entry.result !== null) return entry
      
      // For props
      if (entry.type === "prop" && entry.playerId) {
        const player = game.finalStats
          ?.find(p => p.id === entry.playerId)
        if (!player) return entry
        
        const finalStat = player[entry.market] ?? null
        if (finalStat === null) return entry
        
        const hitOver = finalStat > entry.line
        const hitUnder = finalStat < entry.line
        const push = finalStat === entry.line
        
        let result = "MISS"
        if (push) result = "PUSH"
        else if (entry.direction === "Over" && hitOver) 
          result = "HIT"
        else if (entry.direction === "Under" && hitUnder) 
          result = "HIT"
        
        return { 
          ...entry, 
          result, 
          settledAt: Date.now() 
        }
      }
      
      // For game totals (NCAAB)
      if (entry.type === "game") {
        const total = game.finalAwayScore + 
                      game.finalHomeScore
        const hitOver = total > entry.line
        const hitUnder = total < entry.line
        const push = total === entry.line
        
        let result = "MISS"
        if (push) result = "PUSH"
        else if (entry.direction === "Over" && hitOver)
          result = "HIT"
        else if (entry.direction === "Under" && hitUnder)
          result = "HIT"
        
        return { 
          ...entry, 
          result, 
          settledAt: Date.now() 
        }
      }
      
      return entry
    }))
  }

Call settleNotificationLogs(game) when any game 
status changes to "final".

DAILY RESET

notificationLog resets at start of new calendar day:

  useEffect(() => {
    const today = new Date().toDateString()
    const lastLogDate = localStorage.getItem("lastLogDate")
    
    if (lastLogDate !== today) {
      setNotificationLog([])
      localStorage.setItem("lastLogDate", today)
    }
  }, [])
  
  Use localStorage for lastLogDate only — 
  not for the log entries themselves (in-memory only).
  Log entries reset on page refresh or new day.

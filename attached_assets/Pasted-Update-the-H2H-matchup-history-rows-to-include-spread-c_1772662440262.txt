COVERAGE CALCULATION:
  const determineCoverage = (game) => {
    if (!game.spread || !game.spreadTeam) return "N/A"
    
    const { awayScore, homeScore, spread, spreadTeam } = game
    const margin = awayScore - homeScore
    
    // spread is negative number for favorite
    // e.g. spreadTeam = "HOME", spread = -4.5
    // means home favored by 4.5
    
    if (spreadTeam === "HOME") {
      const adjustedMargin = homeScore - awayScore
      if (adjustedMargin > Math.abs(spread)) return { 
        result: "covered", team: game.homeAbbr 
      }
      if (adjustedMargin === Math.abs(spread)) return { 
        result: "PUSH", team: null 
      }
      return { result: "failed", team: game.homeAbbr }
    }
    
    if (spreadTeam === "AWAY") {
      const adjustedMargin = awayScore - homeScore
      if (adjustedMargin > Math.abs(spread)) return { 
        result: "covered", team: game.awayAbbr 
      }
      if (adjustedMargin === Math.abs(spread)) return { 
        result: "PUSH", team: null 
      }
      return { result: "failed", team: game.awayAbbr }
    }
  }

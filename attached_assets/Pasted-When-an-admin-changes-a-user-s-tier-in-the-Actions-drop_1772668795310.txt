When an admin changes a user's tier in the Actions 
dropdown, automatically charge them the price difference 
via Stripe and grant the correct access for that tier.

---

CURRENT BEHAVIOR
Admin selects a tier from the dropdown → nothing happens 
to Stripe, no charge, no access change.

TARGET BEHAVIOR
Admin selects tier → Stripe charges difference → 
user access updates → UI reflects new tier.

STEP 2 — TIER CHANGE HANDLER

Create a handleTierChange function that fires when 
admin selects a new tier from the dropdown.

  const handleTierChange = async (userId, newTierKey) => {
    
    const user = users.find(u => u.id === userId)
    const currentTier = TIER_PRICES[user.tier]
    const newTier = TIER_PRICES[newTierKey]
    
    // Calculate price difference
    const priceDiff = newTier.pricePerMonth 
                    - currentTier.pricePerMonth

    try {
      const res = await fetch("/api/admin/change-tier", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          userId,
          newTierKey,
          priceDiff,
          newStripePriceId: newTier.stripePriceId
        })
      })
      
      if (!res.ok) throw new Error(await res.text())
      
      const result = await res.json()
      
      // Update local UI state immediately
      setUsers(prev => prev.map(u => 
        u.id === userId 
          ? { ...u, tier: newTierKey } 
          : u
      ))
      
      // Show success toast
      showToast(result.message, "success")
      
    } catch (err) {
      showToast(`Failed: ${err.message}`, "error")
      // Revert dropdown to previous tier in UI
    }
  }

STEP 3 — API ROUTE: /api/admin/change-tier

Create this server-side API route.
Use the Stripe SDK already installed in the project.

  export async function POST(req) {
    const { userId, newTierKey, priceDiff, 
            newStripePriceId } = await req.json()
    
    // 1. Get user from database
    const user = await getUser(userId)
    // Use existing DB/auth method already in codebase
    
    // 2. Handle downgrade to Free
    if (newTierKey === "free") {
      await handleDowngradeToFree(user)
      await updateUserTier(userId, "free")
      return Response.json({ 
        message: "Downgraded to Free. Subscription cancelled." 
      })
    }
    
    // 3. User has no Stripe customer — create one
    if (!user.stripeCustomerId) {
      const customer = await stripe.customers.create({
        email: user.email,
        metadata: { userId: user.id }
      })
      await saveStripeCustomerId(userId, customer.id)
      user.stripeCustomerId = customer.id
    }
    
    // 4. Check if user has active subscription
    const subscriptions = await stripe.subscriptions.list({
      customer: user.stripeCustomerId,
      status: "active",
      limit: 1
    })
    
    const existingSub = subscriptions.data[0]
    
    if (existingSub) {
      // 5a. UPGRADE — update subscription to new price
      // Stripe prorates automatically
      await stripe.subscriptions.update(existingSub.id, {
        items: [{
          id: existingSub.items.data[0].id,
          price: newStripePriceId
        }],
        proration_behavior: "always_invoice",
        // This charges the difference immediately
        payment_behavior: "default_incomplete"
      })
      
    } else {
      // 5b. NEW SUBSCRIPTION — no existing sub
      if (priceDiff > 0) {
        // Charge immediately via payment intent
        // User needs a payment method on file
        const paymentMethods = await stripe.paymentMethods.list({
          customer: user.stripeCustomerId,
          type: "card"
        })
        
        if (!paymentMethods.data.length) {
          return Response.json(
            { error: "No payment method on file for this user." },
            { status: 400 }
          )
        }
        
        // Create subscription which charges immediately
        await stripe.subscriptions.create({
          customer: user.stripeCustomerId,
          items: [{ price: newStripePriceId }],
          default_payment_method: 
            paymentMethods.data[0].id,
          payment_behavior: "default_incomplete",
          expand: ["latest_invoice.payment_intent"]
        })
      }
    }
    
    // 6. Update user tier in database
    await updateUserTier(userId, newTierKey)
    
    // 7. Grant access for new tier
    await grantTierAccess(userId, newTierKey)
    
    return Response.json({
      message: priceDiff > 0
        ? `Upgraded to ${newTierKey}. Charged $${priceDiff} difference.`
        : priceDiff < 0
        ? `Downgraded. Subscription updated.`
        : `Tier updated. No charge difference.`
    })
  }

STEP 4 — DOWNGRADE TO FREE HANDLER

  const handleDowngradeToFree = async (user) => {
    if (!user.stripeCustomerId) return
    
    const subscriptions = await stripe.subscriptions.list({
      customer: user.stripeCustomerId,
      status: "active"
    })
    
    for (const sub of subscriptions.data) {
      await stripe.subscriptions.update(sub.id, {
        cancel_at_period_end: true
        // Cancels at end of billing cycle, not immediately
        // User keeps access until period ends
      })
    }
  }

STEP 5 — GRANT TIER ACCESS FUNCTION

  const grantTierAccess = async (userId, tierKey) => {
    const tier = TIER_PRICES[tierKey]
    
    await updateUser(userId, {
      tier: tierKey,
      playsAllowed: tier.playsAllowed,
      // Reset plays used when upgrading
      playsUsed: tierKey !== "free" ? 0 : undefined,
      accessLevel: tierKey === "free" 
        ? "free" 
        : tierKey === "pro_nba"
        ? "pro_nba"
        : "pro_all",
      updatedAt: new Date().toISOString()
    })
  }

  Use the existing updateUser/database method 
  already in the codebase — do not create a new one.

STEP 7 — DROPDOWN LOADING STATE

While tier change is processing:
  Dropdown shows spinner + "Updating..." text
  Disabled, not clickable
  Delete button also disabled
  
  After success or error:
    Dropdown re-enables
    Shows new tier (success) or reverts (error)
